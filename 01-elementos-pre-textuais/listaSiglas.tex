\begin{siglas}
    \item[CRC] \textit{Cyclic Redundancy Check} (verificação de redundância cíclica)
    \item[EPC] \textit{Electronic Product Code}
    \item[HF] \textit{High Frequency}
    \item[IEC] \textit{International Electrotechnical Commission}
    \item[IoT] \textit{Internet of Things}
    \item[ISO] \textit{International Organization for Standardization}
    \item[NFC] \textit{Near Field Communication}
    \item[RF] Radiofrequência
    \item[RFID] \textit{Radio Frequency Identification}
    \item[RSSI] \textit{Received Signal Strength Indicator}
    \item[TID] \textit{Tag Identifier}
    \item[TTL] \textit{Transistor--Transistor Logic}
    \item[UART] \textit{Universal Asynchronous Receiver--Transmitter}
    \item[UHF] \textit{Ultra High Frequency}
\end{siglas}
